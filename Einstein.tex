\begin{abstract}
The companion paper~\cite{K1} showed that the eigenvalue self-consistency
conditions $\sigma_2^2\,\Box\ln\sigma_i=1$ ($i=1,2$) are algebraically
equivalent to $R_{\mu\nu}=0$ for static, spherically symmetric metrics.
Here we extend this result to the matter-coupled case.  For any metric
$f(r)=1-2m(r)/r$ we prove three exact identities:
$K_1 = 1 - \tfrac{1}{2}r^2 R$,\;
$K_2 = 1 - 8\pi r^2\rho$,\; and
$K_1-K_2 = G_{\theta\theta}/r^2$,
where $R$ is the Ricci scalar, $\rho$ is the energy density, and
$G_{\theta\theta}$ is the angular Einstein tensor component.
The pair $\{K_1,K_2\}$ therefore encodes the complete spherically symmetric
Einstein equations.  As applications we recover the known expressions for
Reissner--Nordstr\"{o}m, Schwarzschild--de~Sitter, and Kerr, and show
that $K_1=1$ holds for \emph{all} black-hole solutions whose $\Delta$ is
at most quadratic in~$r$.
\end{abstract}

\medskip
\noindent\textbf{Keywords:} $K{=}1$ framework $\cdot$ Einstein equations
$\cdot$ eigenvalue self-consistency $\cdot$ spherical symmetry
$\cdot$ matter coupling

\section{Setup and notation}

We work with a general static, spherically symmetric metric
\begin{equation}\label{eq:metric}
  ds^2 = -f(r)\,dt^2 + f(r)^{-1}dr^2 + r^2\,d\Omega^2,
  \qquad f(r)=1-\frac{2m(r)}{r},
\end{equation}
where $m(r)$ is an arbitrary $C^2$ mass function (constant $m=M$
for vacuum, $r$-dependent for matter distributions).
Following~\cite{K1}, define $\Delta(r):=r^2f(r)=r^2-2m(r)r$
and the symplectic eigenvalues
\begin{equation}
  \sigma_1 := r\sqrt{f} = \sqrt{\Delta}/\sqrt{r^2/r^2}
  = \sqrt{\Delta}, \qquad \sigma_2 := r.
\end{equation}
The radial d'Alembertian and the eigenvalue self-consistency
functionals are
\begin{equation}
  \Box\phi := \frac{1}{r^2}\,\partial_r\!\bigl(r^2 f\,\partial_r\phi\bigr),
  \qquad
  K_i := r^2\,\Box\ln\sigma_i \quad (i=1,2).
\end{equation}

% ─────────────────────────────────────────────────────────
\section{Main results}

\begin{theorem}[Exact curvature identities]\label{thm:main}
For metric~\eqref{eq:metric} with $f=1-2m(r)/r$, the following hold
identically for any $C^2$ function $m(r)$:
\begin{align}
  K_1 &= 1 - \tfrac{1}{2}\,r^2\,R, \label{eq:K1R}\\
  K_2 &= 1 - 8\pi r^2\,\rho,       \label{eq:K2rho}\\
  K_1 - K_2 &= 8\pi r^2\,p_{\!\perp},  \label{eq:K12}
\end{align}
where $R=-f''-4f'/r+2(1-f)/r^2$ is the Ricci scalar,
$8\pi\rho := G^t{}_t = 2m'(r)/r^2$ the energy density, and
$8\pi p_{\!\perp}:= G_{\theta\theta}/r^2 = \tfrac{1}{2}f''+2f'/r$
the tangential pressure.
\end{theorem}

\begin{proof}
Write $\Delta = r^2 - 2m(r)r$.  Then $\Delta''=2-2m''(r)r$, and a direct
computation gives
\begin{equation}
  K_1 = \frac{\Delta''}{2}
       = 1 - r\,m''(r) - 2m'(r) + 2m'(r)
       = f + 2rf' + \tfrac{1}{2}r^2 f''.\label{eq:K1exp}
\end{equation}
Comparing with $R = -f''-4f'/r+2(1-f)/r^2$ one checks
$K_1 = 1-\tfrac{1}{2}r^2 R$ (equation~\eqref{eq:K1R}).
For $K_2$: since $\sigma_2=r$,
\begin{equation}
  K_2 = r^2\,\Box\ln r = \frac{1}{r^2}\frac{d}{dr}\!\bigl(r^2 f\cdot\tfrac{1}{r}\bigr)
      \cdot r^2 = \frac{d(rf)}{dr} = f + rf' = 1 - 2m'(r),
\end{equation}
which is~\eqref{eq:K2rho} since $8\pi\rho = 2m'/r^2$.
Equation~\eqref{eq:K12} follows from $G_{\theta\theta}
= \tfrac{1}{2}r^2 f'' + rf' = K_1 - K_2$.
\end{proof}

\begin{corollary}[Complete Einstein encoding]\label{cor:complete}
The pair $(K_1,K_2)$ encodes all three independent components
of the spherically symmetric Einstein equations:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\hline
Einstein component & Expression in $K_i$ & Physical content \\
\hline
$G^t{}_t = 8\pi\rho$
  & $K_2 = 1 - 8\pi r^2\rho$
  & energy density \\
$G^r{}_r = -8\pi p_r$
  & $K_2 - 1 = -8\pi r^2 p_r$ (vacuum)
  & radial pressure \\
$G_{\theta\theta} = 8\pi r^2 p_{\!\perp}$
  & $K_1 - K_2 = 8\pi r^2 p_{\!\perp}$
  & tangential pressure \\
\hline
\end{tabular}
\end{center}
In vacuum ($\rho=p_r=p_{\!\perp}=0$) the conditions $K_1=K_2=1$ recover
the result of~\cite{K1}: $\sigma_2^2\,\Box\ln\sigma_i=1\Leftrightarrow R_{\mu\nu}=0$.
\end{corollary}

\section{Applications}

\subsection*{Black-hole solutions ($\Delta$ at most quadratic in $r$)}

For all metrics with $\Delta(r) = r^2 - 2Mr + c_0$ (constant $c_0$,
covering Kerr with $c_0=a^2$ and Reissner--Nordstr\"{o}m with
$c_0=Q^2$, and Kerr--Newman with $c_0=a^2+Q^2$):
\begin{equation}
  \Delta'' = 2 \quad\Longrightarrow\quad K_1 = 1 \quad \text{exactly.}
\end{equation}
The constant term $c_0$ does not contribute to $\Delta''$; only the
leading $r^2$ coefficient matters.  Thus $K_1=1$ holds for the entire
Kerr--Newman family, independently of mass, spin, and charge.

The deviations in $K_2$ encode the non-Schwarzschild charges:
\begin{align}
  \text{Reissner--Nordstr\"{o}m:}&\quad
    K_2 - 1 = -Q^2/r^2,
    \quad \rho_\mathrm{eff} = Q^2/(8\pi r^4) \;\;\text{[Maxwell]}\\
  \text{Kerr (equatorial):}&\quad
    K_2 - 1 = -4Ma^2r/(r^2+a^2)^2,
    \quad \rho_\mathrm{eff} = Ma^2 r/[2\pi(r^2+a^2)^3]
\end{align}
where $\rho_\mathrm{eff}:=(1-K_2)/(8\pi r^2)$.
For Reissner--Nordstr\"{o}m, $\rho_\mathrm{eff}$ exactly matches
the Maxwell energy density; for Kerr it represents the effective
rotational energy density (vanishing at infinity).

\subsection*{Cosmological constant}

For $\Delta=r^2-2Mr-\Lambda r^4/3$ (Schwarzschild--de~Sitter):
\begin{equation}
  K_1 - 1 = -2\Lambda r^2, \qquad K_2 - 1 = -\Lambda r^2,
\end{equation}
consistent with $\rho_\Lambda = \Lambda/(8\pi)$ and
$p_{\Lambda} = -\Lambda/(8\pi)$ (cosmological constant as vacuum energy).

\subsection*{Physical interpretation of $K_1$}

From $K_1 = 1 - r\,m'' - 2m' + 2m' = 1 - r\,m''(r)$:
\begin{equation}
  K_1 = 1 \quad\Longleftrightarrow\quad m''(r)=0
  \quad\Longleftrightarrow\quad
  \frac{d}{dr}\!\bigl(4\pi r^2\rho\bigr)=0.
\end{equation}
Thus $K_1=1$ means \emph{the mass-shell density $4\pi r^2\rho$ is
constant in $r$}---it holds whenever matter is concentrated at the
origin (a point source), as for all black-hole solutions.
The condition $K_1=1$ is therefore not a statement about the vacuum,
but about the \emph{absence of distributed volume matter}.

\section{Discussion}

Theorem~\ref{thm:main} elevates the two conditions $K_1,K_2$ from
diagnostic tools to exact algebraic coordinates on the space of
spherically symmetric Einstein solutions.  The result clarifies the
scope of the equivalence theorem in~\cite{K1} (which required both
$K_1=K_2=1$, i.e.\ vacuum) and opens two directions:

\medskip
\noindent\textbf{Matter coupling.}
The identities $K_1=1-r^2R/2$ and $K_2=1-8\pi r^2\rho$ hold for
any $C^2$ mass function, including interior stellar solutions.
The eigenvalue self-consistency conditions thus extend naturally
to the matter-coupled sector, with $K_i\neq 1$ measuring the departure
from vacuum.

\medskip
\noindent\textbf{Non-spherical extension.}
In the Kerr case (equatorial plane), $K_1=1$ continues to hold exactly
under the natural generalization $\sigma_1=\sqrt{\Delta}$,
$\sigma_2=\sqrt{r^2+a^2}$ and the area-weighted operator
$\Box_A\phi=(A^{-1})\partial_r(\Delta\,\partial_r\phi)$ with $A=r^2+a^2$.
The deviation $K_2-1=-4Ma^2r/(r^2+a^2)^2$ is interpretable as a
rotational effective energy density.
Whether a null-surface generalization of $\Box_A$ restores
$K_2=1$ for Kerr remains an open problem.
